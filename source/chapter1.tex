\section{命題}
\subsection{命題}
  \begin{Definition}[命題]
    \ \\
ある事柄に対して正しいもしくは誤りであると判断できる主張を文で表したものを命題という．表記を簡潔にするため，記号$:$を用いて\\
\centerline{$P:12$は$2$で割り切れる}
と表す．
\end{Definition}
    \begin{Definition}[真と偽]
    \ \\
　命題が正しい場合，真理値が真（True）の命題であるといい，偽（False）である場合は真理値が偽の命題であるという．
  \end{Definition}
  \subsection{合成命題}
  \begin{Definition}[合成命題]
    \ \\
　いくつかの命題を組み合わせた，あるいは否定した命題を，合成命題あるいは複合命題という．これに対し，元の命題を単純命題という．合成命題も真偽の判定ができるものでなければならない．
  \end{Definition}  
  \begin{Definition}[連言命題]
    \ \\
　命題$p,q$に対して，$p$および$q$のいずれも真であるときのみ真となり，他の場合に偽となる命題論理を連言命題あるいは論理積といい，$p\land q$と表す．
  \end{Definition}
  \begin{Definition}[包含的選言命題]
    \ \\
　命題$p,q$に対して，$p$および$q$のどちらかが真であるとき真となり，他の場合に偽となる合成命題を包含的選言命題といい，$p\lor q$と表す．
  \end{Definition} 
   \begin{Definition}[排他的選言命題]
    \ \\
　命題$p,q$に対して，$p$あるいは$q$のどちらかのみが真であり，他方が偽であるとき真となり，他の場合に偽となる合成命題を包含的選言命題といい，$p \underline{\lor} q$と表す．
  \end{Definition}
  \subsection{否定}
  \begin{Definition}[否定]
  \ \\
　命題$P$に対して，$P$が真のとき偽，$P$が偽のとき真となる命題を否定命題といい，$ \sim P$または$\overline{P}$と示す．
　\end{Definition} 
 \begin{Theorem}[ド・モルガンの法則]
 \ \\
　任意の命題$P,Q$に対し次の法則が成り立つ．
　\begin{align*}
　&\sim(P \lor Q)=(\sim P) \land (\sim Q)&\\
　&\sim(P \land Q)=(\sim P) \lor (\sim Q)&
　\end{align*}
 \end{Theorem}   
   \begin{Definition}[多重否定]
  \ \\
　否定命題を否定した命題を二重否定命題といい，$ \sim(\sim P)$または$\overline{\overline{P}}$と示す．その真偽は元の命題$P$の真偽に等しい．
  \end{Definition}
\begin{Theorem}
\ \\
$p,q$を命題とすると，次が成り立つ．
\begin{enumerate}
 \item $p \to q \equiv (\sim p) \lor q$
 \item $p \to q \equiv (\sim q) \to (\sim p)$
 \item $\sim (p \to q) \equiv p \land (\sim q)$
\end{enumerate}
\end{Theorem}

  \ \\

  \end{Definition}

  \subsection{命題の同値関係}
  \begin{Definition}[同値]
  \ \\
　2 つの合成命題の真理表が一致するとき，同値であるといい，記号$\equiv$で結ぶ。
  \end{Definition}  
  \section{条件文}
  \subsection{条件文}
    \begin{Definition}[条件文]
  \ \\
  　2つの命題$p$および$q$を，「$p$ならば$q$である」という形式で結んだ文を条件文という．このとき$p$を仮定，$q$を結論といい，$p \to q$ と表す．
  \end{Definition} 
 \begin{Theorem}[条件文の同値関係]
 \ \\
 　「$p$ならば$q$である」という条件文は，「$p$でないか，または$q$である」という命題と同値である。すなわち，$p \to q \equiv  (\sim p) \lor q$となる．
 \end{Theorem}
\begin{Proof}
\ \\
条件文$p \to q$は$\sim(p \land \sim q)$である．\\
ド・モルガンの定理より，$(\sim p)\lor (\sim(\sim q))$\\
よって，$p \to q \equiv (\sim p) \lor q$
\end{Proof}
 \begin{exam}[条件文の否定]
  \ \\
 　「$p$ならば$q$である」という条件文の否定は，$\sim((\sim p)\lor q) \equiv p \land (\sim q)$であるから，「$p$であって$q$でない」となる．
 \end{exam}
 \begin{Definition}[双条件文]
  \ \\
 　「$p$ならば$q$であり，かつ$q$ならば$p$である」という条件文を双条件文といい$p \leftrightarrow q$と表す．
 \end{Definition}
 \subsection{条件文の逆・裏・対偶}
 \begin{Definition}[条件文の逆]
  \ \\
 　「$p$ならば$q$で
ある」という条件文に対して，「$q$ならば$p$である」という条件文をもとの条件文の逆といい，$p \to q$と表す．
  \end{Definition}  
  \begin{Definition}[条件文の裏]
  \ \\
 　「$p$ならば$q$である」という条件文に対して，「$p$でないならば$q$でない」という条件文をもとの条件文の裏といい，$\sim p \to \sim q$と表す．
  \end{Definition}  
  \begin{Definition}[条件文の対偶]
  \ \\
 　「$p$ならば$q$である」という条件文に対して，「$q$でないならば$p$でない」という条件文をもとの条件文の対偶といい，$\sim q \to \sim p$と表す．
  \end{Definition}  
  \begin{exam}
  \ \\
 　条件文$p,q$の逆，裏および対偶は次のようになる．

    $ T $ & $ T $ & $ T $ & $ T $ & $ T $ & $ T $ \\

\end{exam}
\begin{Theorem}[対偶の同値性]
\ \\
　条件文とその対偶のどちらかが真である場合，他方もまた真である．
　\end{Theorem}
　\begin{Proof}
　\ \\
　真理表より明らか．
　\end{Proof}
　 \subsection{必要条件と十分条件}
  \begin{Definition}[必要条件と十分条件]
  \ \\
 　$p \to q$であるとき，$q$を$p$であるための必要条件，$p$を$q$であるための十分条件という．
  \end{Definition}  　 
  \begin{Definition}[必要十分条件]
  \ \\
 　$p \to q \land q \to p$であるとき，$q$を$p$であるための必要十分条件といい$p \rightleftharpoons q$で表す．
  \end{Definition}  　 
  \subsection{真理集合}
  \begin{Definition}[命題関数]
  \ \\ 
　集合$X$の元$x\in X$に対して，その真偽が判定できるような条件$p$ が与えられたとき，$X$を定義域，値域を  \{ T, F \} とする命題関数という。
　\end{Definition}
  \subsection{全称命題と特称命題}
    \begin{Definition}[全称命題]
  \ \\ 
　集合$X$の元$x\in X$に対して，その真偽が判定できるような条件$p$ が与えられたとき，$X$を定義域，値域を  \{ T, F \} とする命題関数という。
　\end{Definition}
